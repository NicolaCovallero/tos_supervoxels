121 tos_supervoxels::addSupervoxelConnectionsToViewer (pcl::PointXYZRGBA &supervoxel_center,
122  pcl::PointCloud<pcl::PointXYZRGBA>& adjacent_supervoxel_centers,
123  std::string supervoxel_name,
124  boost::shared_ptr<pcl::visualization::PCLVisualizer> & viewer)
125 {
126  vtkSmartPointer<vtkPoints> points = vtkSmartPointer<vtkPoints>::New ();
127  vtkSmartPointer<vtkCellArray> cells = vtkSmartPointer<vtkCellArray>::New ();
128  vtkSmartPointer<vtkPolyLine> polyLine = vtkSmartPointer<vtkPolyLine>::New ();
129 
130  //Iterate through all adjacent points, and add a center point to adjacent point pair
131  pcl::PointCloud<pcl::PointXYZRGBA>::iterator adjacent_itr = adjacent_supervoxel_centers.begin ();
132  for ( ; adjacent_itr != adjacent_supervoxel_centers.end (); ++adjacent_itr)
133  {
134  points->InsertNextPoint (supervoxel_center.data);
135  points->InsertNextPoint (adjacent_itr->data);
136  }
137  // Create a polydata to store everything in
138  vtkSmartPointer<vtkPolyData> polyData = vtkSmartPointer<vtkPolyData>::New ();
139  // Add the points to the dataset
140  polyData->SetPoints (points);
141  polyLine->GetPointIds ()->SetNumberOfIds(points->GetNumberOfPoints ());
142  for(unsigned int i = 0; i < points->GetNumberOfPoints (); i++)
143  polyLine->GetPointIds ()->SetId (i,i);
144  cells->InsertNextCell (polyLine);
145  // Add the lines to the dataset
146  polyData->SetLines (cells);
147  viewer->addModelFromPolyData (polyData,supervoxel_name);
148 }

150 void tos_supervoxels::detectObjectsOnTable(pcl::PointCloud<pcl::PointXYZRGBA>::Ptr cloud, double zmin, double zmax, pcl::PointIndices::Ptr objectIndices, bool filter_input_cloud)
151 {
152  // Objects for storing the point clouds.
153  pcl::PointCloud<pcl::PointXYZRGBA>::Ptr plane(new pcl::PointCloud<pcl::PointXYZRGBA>);
154  pcl::PointCloud<pcl::PointXYZRGBA>::Ptr convexHull(new pcl::PointCloud<pcl::PointXYZRGBA>);
155 
156  // Get the plane model, if present.
157  pcl::ModelCoefficients::Ptr coefficients(new pcl::ModelCoefficients);
158  pcl::SACSegmentation<pcl::PointXYZRGBA> segmentation;
159  segmentation.setInputCloud(cloud);
160  segmentation.setModelType(pcl::SACMODEL_PLANE);
161  segmentation.setMethodType(pcl::SAC_RANSAC);
162  segmentation.setDistanceThreshold(0.01);
163  segmentation.setOptimizeCoefficients(true);
164  pcl::PointIndices::Ptr planeIndices(new pcl::PointIndices);
165  segmentation.segment(*planeIndices, *coefficients);
166 
167  if (planeIndices->indices.size() == 0)
168  std::cout << "Could not find a plane in the scene." << std::endl;
169  else
170  {
171  // Copy the points of the plane to a new cloud.
172  pcl::ExtractIndices<pcl::PointXYZRGBA> extract;
173  extract.setInputCloud(cloud);
174  extract.setIndices(planeIndices);
175  extract.filter(*plane);
176 
177  // Retrieve the convex hull.
178  pcl::ConvexHull<pcl::PointXYZRGBA> hull;
179  hull.setInputCloud(plane);
180  // Make sure that the resulting hull is bidimensional.
181  hull.setDimension(2); //2dimension -> planar convex hull
182  hull.reconstruct(*convexHull);
183 
184  // Redundant check.
185  if (hull.getDimension() == 2)
186  {
187  // Prism object.
188  pcl::ExtractPolygonalPrismData<pcl::PointXYZRGBA> prism;
189  prism.setInputCloud(cloud);
190  prism.setInputPlanarHull(convexHull);
191  // First parameter: minimum Z value. Set to 0, segments objects lying on the plane (can be negative).
192  // Second parameter: maximum Z value[meters], set to 10cm. Tune it according to the height of the objects you expect.
193  prism.setHeightLimits(zmin, zmax);
194 
195  //std::cout << "size of objectIndices before segmentation: " << objectIndices.indices.size() << std::endl;
196  //pcl::PointIndices::Ptr objectIndices(new pcl::PointIndices);
197  prism.segment(*objectIndices);
198  //std::cout << "size of objectIndices AFTER segmentation: " << objectIndices.indices.size() << std::endl;
199 
200  // Get and show all points retrieved by the hull.
201  extract.setIndices(objectIndices);
202 
203  if(filter_input_cloud)
204  extract.filter(*cloud);
205 
206  }
207  else std::cout << "The chosen hull is not planar." << std::endl;
208  }
209 }

243 
244  if(!this->initialized)
245  {

248  }

252  //float elapsed_time_plane = (cv::getTickCount()-t_ini)/cv::getTickFrequency();
253 
254  if(this->seed_resolution < 0.013)
255  pcl::console::print_warn("[tos_supervoxels] seed resolution very low, the segmentation could be fragmented.");
256 
257  pcl::SupervoxelClustering<pcl::PointXYZRGBA> super (this->voxel_resolution, this->seed_resolution);
258 
259  super.setInputCloud (this->cloud);
260 
261  //super.setNormalCloud (input_normals);
262 
263  super.setColorImportance (this->color_importance);
264  super.setSpatialImportance (this->spatial_importance);
265  super.setNormalImportance (this->normal_importance);
266 
267  super.extract (supervoxel_clusters);
268  labeled_voxel_cloud = super.getLabeledVoxelCloud ();
269  pcl::PointCloud<pcl::PointXYZRGBA>::Ptr voxel_centroid_cloud = super.getVoxelCentroidCloud ();
270  sv_normal_cloud = super.makeSupervoxelNormalCloud (supervoxel_clusters);
271  pcl::PointCloud<pcl::PointXYZL>::Ptr full_labeled_cloud = super.getLabeledCloud ();
272 
273  super.getSupervoxelAdjacency (supervoxel_adjacency);
274 
275  std::map <uint32_t, pcl::Supervoxel<pcl::PointXYZRGBA>::Ptr > refined_supervoxel_clusters;
276 
277  // NO REFINEMENT!!
278  //super.refineSupervoxels (3, refined_supervoxel_clusters);
279 
280  pcl::PointCloud<pcl::PointXYZL>::Ptr refined_labeled_voxel_cloud = super.getLabeledVoxelCloud ();
281  pcl::PointCloud<pcl::PointNormal>::Ptr refined_sv_normal_cloud = super.makeSupervoxelNormalCloud (refined_supervoxel_clusters);
282  pcl::PointCloud<pcl::PointXYZL>::Ptr refined_full_labeled_cloud = super.getLabeledCloud ();

284  typedef boost::adjacency_list<boost::setS, boost::setS, boost::undirectedS, uint32_t, float> VoxelAdjacencyList;
285  VoxelAdjacencyList supervoxel_adjacency_list;
286  super.getSupervoxelAdjacencyList (supervoxel_adjacency_list);
287 
288  // Segmentation Stuff
289 
290  uint k_factor = 0;
291  if (use_extended_convexity)
292  k_factor = 1;
293 
294  pcl::LCCPSegmentation<pcl::PointXYZRGBA> lccp;
295  lccp.setConcavityToleranceThreshold (this->concavity_tolerance_threshold);
296  lccp.setSanityCheck (this->use_sanity_criterion);
297  lccp.setSmoothnessCheck (true, this->voxel_resolution, this->seed_resolution, this->smoothness_threshold);
298  lccp.setKFactor (k_factor);
299  lccp.setInputSupervoxels (this->supervoxel_clusters, this->supervoxel_adjacency);
300  lccp.setMinSegmentSize (this->min_segment_size);
301  lccp.segment ();
302 
303  pcl::PointCloud<pcl::PointXYZL>::Ptr sv_labeled_cloud = super.getLabeledCloud ();
304  lccp_labeled_cloud = sv_labeled_cloud->makeShared ();
305  lccp.relabelCloud (*lccp_labeled_cloud);
306  //std::cout << "size lccp_labeled_cloud: "<<lccp_labeled_cloud->points.size() << "\n";
307  //---------------------------------------------------------------------------------------------
308 
309  //float elapsed_time_seg = (cv::getTickCount()-t_ini)/cv::getTickFrequency() - elapsed_time_plane;
310 
311  //------------------------------ SEGMENTATION POST PROCESSING ---------------------------------
312  //construct for each segmented object the point cloud associated to that object
313  for (int i = 0; i < lccp_labeled_cloud->points.size(); ++i)
314  {
315 
316  uint32_t idx = lccp_labeled_cloud->points.at(i).label;
317 
318  //in this way we enlarges the vector everytime we encounter a greater label. So we don't need to pass all
319  // labeeld point cloud to see what is the greater label, and then to resize the vector.
320  if(idx >= detected_objects.size()) // keep in mind that there is also the label 0!
321  detected_objects.resize(idx+1);
322 
323  pcl::PointXYZRGBA tmp_point_rgb;
324  tmp_point_rgb = cloud->points.at(i);
325  detected_objects[idx].object_cloud.points.push_back(tmp_point_rgb);
326 
327  detected_objects[idx].label = (int)idx;
328  }
329 
330  //remove segments with too few points
331  // it will removes te ones with few points or the ones with no points (these are created because of the labels of lccp)
332  int size = detected_objects.size();
333  int i = 0;
334  while (i < size)
335  {
336  if(detected_objects[i].object_cloud.size() < this->th_points)
337  {
338  detected_objects.erase(detected_objects.begin() + i);
339  size = detected_objects.size();
340  }
341  else
342  i++;
343  }
344 
345  //float elapsed_time = (cv::getTickCount()-t_ini)/cv::getTickFrequency();
346 
347  //std::cout << "elapsed_time (segmentation and postprocessing): " << elapsed_time << "\nelapsed_time of LCCP segmentation: " << elapsed_time_seg << std::endl;
348  //std::cout << "elapsed_time plane estimation: " << elapsed_time_plane << std::endl;
349 
350  return;
351 }

353 void tos_supervoxels::show_super_voxels(boost::shared_ptr<pcl::visualization::PCLVisualizer> & viewer,
354  bool show_adjacency_map,
355  bool show_super_voxel_normals)
356 {
357  if(this->detected_objects.size() > 0)
358  {
359  viewer->addPointCloud (this->labeled_voxel_cloud, "supervoxel_cloud");
360 
361  //pcl::visualization::PointCloudColorHandlerRGBField<pcl::PointXYZRGBA> rgb_cloud_obj(cloud);
362  //viewer->addPointCloud<pcl::PointXYZRGBA> (cloud, rgb_cloud_obj, "maincloud");

364  viewer->addPointCloudNormals<pcl::PointNormal> (this->sv_normal_cloud,1,0.05f, "supervoxel_normals");

366  if(show_adjacency_map)
367  {
368  std::multimap<uint32_t,uint32_t>::iterator label_itr = (this->supervoxel_adjacency).begin ();
369  for ( ; label_itr != (this->supervoxel_adjacency).end(); )
370  {
371  //First get the label
372  uint32_t supervoxel_label = label_itr->first;
373  //Now get the supervoxel corresponding to the label
374 
375  pcl::Supervoxel<pcl::PointXYZRGBA>::Ptr supervoxel = this->supervoxel_clusters.at (supervoxel_label);
376 
377  //Now we need to iterate through the adjacent supervoxels and make a point cloud of them
378  pcl::PointCloud<pcl::PointXYZRGBA> adjacent_supervoxel_centers;
379  std::multimap<uint32_t,uint32_t>::iterator adjacent_itr = this->supervoxel_adjacency.equal_range (supervoxel_label).first;
380  for ( ; adjacent_itr!=this->supervoxel_adjacency.equal_range (supervoxel_label).second; ++adjacent_itr)
381  {
382  pcl::Supervoxel<pcl::PointXYZRGBA>::Ptr neighbor_supervoxel = this->supervoxel_clusters.at (adjacent_itr->second);
383  adjacent_supervoxel_centers.push_back (neighbor_supervoxel->centroid_);
384  }
385  //Now we make a name for this polygon
386  std::stringstream ss;
387  ss << "supervoxel_" << supervoxel_label;
388  //This function is shown below, but is beyond the scope of this tutorial - basically it just generates a "star" polygon mesh from the points given
389  addSupervoxelConnectionsToViewer (supervoxel->centroid_, adjacent_supervoxel_centers, ss.str (), viewer);
390  //Move iterator forward to next label
391  label_itr = (this->supervoxel_adjacency).upper_bound (supervoxel_label);
392  }
393  }
394 
395  }
396  return;
